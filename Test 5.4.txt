#include <stdio.h>

int* findMinMax(int arr[], int n)
{
    static int minMax[2];
    int i;

    minMax[0] = minMax[1] = arr[0];

    for (i = 1; i < n; i++) 
    {
        if (arr[i] < minMax[0])
            minMax[0] = arr[i];
        if (arr[i] > minMax[1])
            minMax[1] = arr[i];
    }

    return minMax;
}

int main() 
{
    int n, i, *minMax;
    int arr[100];

    printf("Number of values you want to input: ");
    scanf("%d", &n);

    printf("Input %d values\n", n);
    for (i = 0; i < n; i++) 
    {
        scanf("%d", &arr[i]);
    }

    minMax = findMinMax(arr, n);
    printf("Minimum value is: %d\n", minMax[0]);
    printf("Maximum value is: %d\n", minMax[1]);

    return 0;
}

OUTPUT:

Number of values you want to input: 5
Input 5 values
25 11 35 65 20
Minimum value is: 11
Maximum value is: 65
